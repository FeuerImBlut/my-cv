function indexEqualsValue(a) {
    let result = -1;
    let theCycle = (begin, end, result, a) => {
        if (begin > end) {
            return result;
        } else {
            let mid = Math.floor(begin - ((begin - end) / 2));
            if (a[mid] == mid) {
                result = mid;
                return theCycle(begin, mid - 1, result, a);
            } else if (a[mid] > mid) {
                return theCycle(begin, mid - 1, result, a);
            } else {
                return theCycle(mid + 1, end, result, a);
            }
        }
    }
    return theCycle(0, a.length - 1, result, a);
}
